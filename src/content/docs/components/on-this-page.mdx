## Installation
```bash
npx shadcn@latest add http://ui.elixirlabs.in/registry/on-this-page.json
```

## Usage

Import the component and provide sections with items:

```tsx
import { OnThisPage } from "@/registry/on-this-page";
```


```tsx
<OnThisPage
  sections={[
    {
      items: [
        { title: "Introduction", href: "#introduction" },
        { title: "information", href: "#unformation" },
        { title: "info", href: "#info" },
      ],
    },
  ]}
/>
```
